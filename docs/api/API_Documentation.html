    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({phone})
  });

    method: 'POST',
    headers: {'Content-Type': 'application/json'},

import requests

class CartWiseClient:
    def __init__(self, base_url='http://localhost:8002'):
        self.base_url = base_url
        self.token = None

    def request_otp(self, phone):
        response = requests.post(
            f'{self.base_url}/auth/request-otp',
            json={'phone': phone}
        )
        return response.json()

    def verify_otp(self, phone, otp_code):
        response = requests.post(
            f'{self.base_url}/auth/verify-otp',
            json={'phone': phone, 'otp_code': otp_code}
        )
        data = response.json()
        self.token = data['auth_token']
        return data

    def assign_cart(self, phone):
        response = requests.post(
            f'{self.base_url}/carts/assign',
            headers={'Authorization': f'Bearer {self.token}'},
            json={'phone': phone, 'otp_code': '0000'}
        )
        return response.json()

# Usage
client = CartWiseClient()
client.request_otp('0501234567')
client.verify_otp('0501234567', '1234')
cart = client.assign_cart('0501234567')
print(f"Cart {cart['cart']['cart_id']} assigned!")
